# M5 Web UI Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the Vue 3.5 SPA (`apps/web`) with all 10 mise screens — Dashboards, Graph Explorer, Review Workbench, Findings, Resolution, Audit Q&A chat, Timeline, Notifications, Reports, Corpus Admin — consuming `@mise/contract` types, behind OIDC, with no model client in the browser.

**Architecture:** Composable-first Vue 3.5 SPA. Views (< 400 lines) compose child components + `useX()` composables. A typed fetch wrapper (`api/client.ts`) injects bearer tokens from a Pinia auth store. SSE for the Q&A chat uses POST-based `ReadableStream` (not `EventSource`). All types come from the `@mise/contract` workspace package — never hand-mirrored.

**Tech Stack:** Vue 3.5 · Vite 8 · TypeScript 6.0 · Tailwind 4 · reka-ui/CVA · Vue Flow + elkjs · TanStack Vue Table · oidc-client-ts · Pinia · Vitest 4 + @vue/test-utils + happy-dom · pnpm 11

## Global Constraints

- Every `.vue` file ≤ 400 lines; line length = 80 (Prettier)
- SFC order: `<script setup>` → `<template>` → `<style>`
- PascalCase filenames for `.vue`; `useX.ts` for composables; kebab-case for plain `.ts`
- `ref`/`computed` over `reactive` for primitives; Pinia only for auth + notifications
- Types from `@mise/contract` only — never hand-mirror types from the Go API
- Tier comes from the server bootstrap call — never client-asserted
- The browser never calls a model provider; no model SDK imports in `apps/web/`
- All review/resolution write endpoints send `Idempotency-Key`
- Translate toggle disabled + refusal when `!translate_allowed` — no API call made
- Webhooks disabled/internal-only until DECISIONS 19 closes
- Tests use Vitest 4 + @vue/test-utils + happy-dom; mock `fetch`, never call real APIs
- Commit messages: no AI attribution trailers

```typescript
import type { ApiProblem } from '@mise/contract';

const BASE_URL =
  (typeof import.meta !== 'undefined' ? import.meta.env?.VITE_API_URL : undefined) ?? '/api/v1';

let bearerToken: string | null = null;

export function setBearerToken(t: string | null): void {
  bearerToken = t;
}

export class ApiClientError extends Error {
  constructor(public readonly problem: ApiProblem) {
    super(problem.title);
    this.name = 'ApiClientError';
  }
}

async function request<T>(method: string, path: string, body?: unknown): Promise<T> {
  const headers: Record<string, string> = {
    Accept: 'application/json',
  };
  if (bearerToken) {
    headers['Authorization'] = `Bearer ${bearerToken}`;
  }
  if (body !== undefined) {
    headers['Content-Type'] = 'application/json';
  }
  const res = await fetch(`${BASE_URL}${path}`, {
    method,
    headers,
    body: body !== undefined ? JSON.stringify(body) : undefined,
  });
  if (!res.ok) {
    const problem = (await res.json()) as ApiProblem;
    throw new ApiClientError(problem);
  }
  if (res.status === 204) {
    return undefined as T;
  }
  return (await res.json()) as T;
}

export function apiGet<T>(path: string, params?: Record<string, string>): Promise<T> {
  const qs = params ? '?' + new URLSearchParams(params).toString() : '';
  return request<T>('GET', `${path}${qs}`);
}

export function apiPost<T>(path: string, body?: unknown): Promise<T> {
  return request<T>('POST', path, body);
}

export function apiDelete(path: string): Promise<void> {
  return request<void>('DELETE', path);
}
```

```typescript
import type { NavigationGuardNext, RouteLocationNormalized } from 'vue-router';
import { useAuthStore } from './store.js';

const PUBLIC_ROUTES = ['/login', '/callback'];

export function authGuard(
  to: RouteLocationNormalized,
  _from: RouteLocationNormalized,
  next: NavigationGuardNext,
): void {
  if (PUBLIC_ROUTES.includes(to.path)) {
    next();
    return;
  }
  const store = useAuthStore();
  if (!store.isAuthenticated) {
    next('/login');
    return;
  }
  if (to.path === '/admin' && !store.capabilities.admin_allowed) {
    next('/');
    return;
  }
  next();
}
```

- [ ] **Step 2: Create `apps/web/src/composables/useTranslate.ts`**

```typescript
import { ref, computed } from 'vue';
import { useAuthStore } from '@/auth/store.js';
import { apiPost } from '@/api/client.js';
import type { TranslateResponse } from '@mise/contract';

const cache = new Map<string, string>();

export function useTranslate() {
  const store = useAuthStore();
  const isTranslating = ref(false);
  const canTranslate = computed(() => store.capabilities.translate_allowed);

  async function translate(text: string, sourceLang: string, targetLang: string): Promise<string> {
    if (!canTranslate.value) {
      throw new Error('Translation not permitted');
    }
    const key = `${sourceLang}:${targetLang}:${text}`;
    const cached = cache.get(key);
    if (cached) return cached;

    isTranslating.value = true;
    try {
      const res = await apiPost<TranslateResponse>('/translate', {
        text,
        source_lang: sourceLang,
        target_lang: targetLang,
      });
      cache.set(key, res.translated_text);
      return res.translated_text;
    } finally {
      isTranslating.value = false;
    }
  }

  return { translate, isTranslating, canTranslate };
}
```

```typescript
export type SseCallbacks = {
  onEvent: (event: string, data: string) => void;
  onDone: () => void;
  onError: (err: Error) => void;
};

export async function parseSseStream(
  reader: ReadableStreamDefaultReader<Uint8Array>,
  callbacks: SseCallbacks,
): Promise<void> {
  const decoder = new TextDecoder();
  let buffer = '';
  let currentEvent = '';

  try {
    for (;;) {
      const { done, value } = await reader.read();
      if (done) break;
      buffer += decoder.decode(value, { stream: true });
      const lines = buffer.split('\n');
      buffer = lines.pop() ?? '';
      for (const line of lines) {
        if (line.startsWith('event:')) {
          currentEvent = line.slice(6).trim();
        } else if (line.startsWith('data:')) {
          const data = line.slice(5).trim();
          if (currentEvent) {
            callbacks.onEvent(currentEvent, data);
            currentEvent = '';
          }
        }
      }
    }
    callbacks.onDone();
  } catch (err) {
    if (err instanceof DOMException && err.name === 'AbortError') {
      callbacks.onDone();
    } else {
      callbacks.onError(err instanceof Error ? err : new Error(String(err)));
    }
  }
}
```

```typescript
import { ref, onUnmounted } from 'vue';
import { parseSseStream } from '@/api/sse.js';
import { useAuthStore } from '@/auth/store.js';
import type {
  SseCitationEvent,
  SseChainEvent,
  SseEvidenceCheckedEvent,
  SseDoneEvent,
  SseErrorEvent,
} from '@mise/contract';

const REASONING_URL =
  (typeof import.meta !== 'undefined' ? import.meta.env?.VITE_REASONING_URL : undefined) ??
  '/reasoning';

export function useChat() {
  const question = ref('');
  const tokens = ref('');
  const citations = ref<SseCitationEvent['data'][]>([]);
  const chain = ref<SseChainEvent['data'][]>([]);
  const evidenceChecked = ref<SseEvidenceCheckedEvent['data'][]>([]);
  const abstain = ref<string | null>(null);
  const done = ref<SseDoneEvent['data'] | null>(null);
  const error = ref<SseErrorEvent['data'] | null>(null);
  const isStreaming = ref(false);

  let abortController: AbortController | null = null;

  function reset(): void {
    tokens.value = '';
    citations.value = [];
    chain.value = [];
    evidenceChecked.value = [];
    abstain.value = null;
    done.value = null;
    error.value = null;
  }

  function handleEvent(event: string, raw: string): void {
    const data = JSON.parse(raw) as Record<string, unknown>;
    switch (event) {
      case 'token':
        tokens.value += (data as { text: string }).text;
        break;
      case 'citation':
        citations.value = [...citations.value, data as SseCitationEvent['data']];
        break;
      case 'chain':
        chain.value = [...chain.value, data as SseChainEvent['data']];
        break;
      case 'evidence_checked':
        evidenceChecked.value = [...evidenceChecked.value, data as SseEvidenceCheckedEvent['data']];
        break;
      case 'abstain':
        abstain.value = (data as { reason: string }).reason;
        break;
      case 'done':
        done.value = data as SseDoneEvent['data'];
        break;
      case 'error':
        error.value = data as SseErrorEvent['data'];
        break;
    }
  }

  async function send(): Promise<void> {
    if (isStreaming.value || !question.value.trim()) return;
    reset();
    isStreaming.value = true;
    abortController = new AbortController();
    const store = useAuthStore();

    try {
      const headers: Record<string, string> = {
        'Content-Type': 'application/json',
      };
      if (store.token) {
        headers['Authorization'] = `Bearer ${store.token}`;
      }
      const res = await fetch(`${REASONING_URL}/chat`, {
        method: 'POST',
        headers,
        body: JSON.stringify({
          question: question.value,
        }),
        signal: abortController.signal,
      });
      if (!res.ok || !res.body) {
        error.value = {
          type: 'stream_error',
          detail: `HTTP ${res.status}`,
        };
        return;
      }
      await parseSseStream(res.body.getReader(), {
        onEvent: handleEvent,
        onDone: () => {},
        onError: (err) => {
          error.value = {
            type: 'stream_error',
            detail: err.message,
          };
        },
      });
    } catch (err) {
      if (err instanceof DOMException && err.name === 'AbortError') {
        return;
      }
      error.value = {
        type: 'stream_error',
        detail: err instanceof Error ? err.message : String(err),
      };
    } finally {
      isStreaming.value = false;
      abortController = null;
    }
  }

  function stop(): void {
    abortController?.abort();
  }

  onUnmounted(() => stop());

  return {
    question,
    tokens,
    citations,
    chain,
    evidenceChecked,
    abstain,
    done,
    error,
    isStreaming,
    send,
    stop,
  };
}
```

```typescript
function makeStream(chunks: string[]): ReadableStreamDefaultReader<Uint8Array> {
  const encoder = new TextEncoder();
  let i = 0;
  const stream = new ReadableStream<Uint8Array>({
    pull(controller) {
      if (i < chunks.length) {
        controller.enqueue(encoder.encode(chunks[i]));
        i++;
      } else {
        controller.close();
      }
    },
  });
  return stream.getReader();
}
```

```typescript
import { defineStore } from 'pinia';
import { ref, computed } from 'vue';
import { apiGet, apiPost } from '@/api/client.js';
import type { CursorPage, Notification } from '@mise/contract';

export const useNotificationStore = defineStore('notifications', () => {
  const items = ref<Notification[]>([]);
  const unreadCount = computed(() => items.value.filter((n) => !n.read).length);

  async function load(): Promise<void> {
    const page = await apiGet<CursorPage<Notification>>('/notifications');
    items.value = [...page.items];
  }

  async function markRead(id: string): Promise<void> {
    await apiPost(`/notifications/${id}/read`);
    const item = items.value.find((n) => n.id === id);
    if (item) {
      const idx = items.value.indexOf(item);
      items.value = [
        ...items.value.slice(0, idx),
        { ...item, read: true },
        ...items.value.slice(idx + 1),
      ];
    }
  }

  return { items, unreadCount, load, markRead };
});
```

```vue
import { ref } from 'vue';
import { useAuthStore } from '@/auth/store.js';
import { Download } from 'lucide-vue-next';

const BASE_URL =
  (typeof import.meta !== 'undefined' ? import.meta.env?.VITE_API_URL : undefined) ?? '/api/v1';

const loading = ref<string | null>(null);

async function download(path: string, filename: string): Promise<void> {
  loading.value = path;
  const store = useAuthStore();
  try {
    const headers: Record<string, string> = {};
    if (store.token) {
      headers['Authorization'] = `Bearer ${store.token}`;
    }
    const res = await fetch(`${BASE_URL}${path}`, {
      headers,
    });
    const blob = await res.blob();
    const url = URL.createObjectURL(blob);
    const a = document.createElement('a');
    a.href = url;
    a.download = filename;
    a.click();
    URL.revokeObjectURL(url);
  } finally {
    loading.value = null;
  }
}
</script>

<template>

    <button
      :disabled="loading !== null"
```

```typescript
import { ref, computed } from 'vue';
import { apiGet } from '@/api/client.js';
import type { CursorPage } from '@mise/contract';

export function usePagination<T>(path: string) {
  const items = ref<T[]>([]) as { value: T[] };
  const loading = ref(false);
  const cursor = ref<string | null>(null);
  const hasMore = computed(() => cursor.value !== null);
  const initialized = ref(false);

  async function loadMore(): Promise<void> {
    loading.value = true;
    try {
      const params: Record<string, string> = {};
      if (cursor.value) {
        params['cursor'] = cursor.value;
      }
      const page = await apiGet<CursorPage<T>>(path, params);
      items.value = [...items.value, ...page.items];
      cursor.value = page.cursor;
      initialized.value = true;
    } finally {
      loading.value = false;
    }
  }

  return { items, loading, hasMore, loadMore, initialized };
}
```

```typescript
import { describe, expect, it } from 'vitest';

const PRIVATE_RANGES = ['10.0.0.0', '192.168.1.1', '127.0.0.1', '172.16.0.1', '[::1]'];

function isPrivateAddress(url: string): boolean {
  try {
    const parsed = new URL(url);
    const host = parsed.hostname;
    if (host === 'localhost' || host === '[::1]') return true;
    if (host.startsWith('10.')) return true;
    if (host.startsWith('192.168.')) return true;
    if (host.startsWith('127.')) return true;
    if (host.startsWith('172.')) {
      const second = parseInt(host.split('.')[1] ?? '0');
      if (second >= 16 && second <= 31) return true;
    }
    return false;
  } catch {
    return true;
  }
}

describe('webhook egress validation', () => {
  for (const addr of PRIVATE_RANGES) {
    it(`rejects private address ${addr}`, () => {
      expect(isPrivateAddress(`http://${addr}/hook`)).toBe(true);
    });
  }

  it('allows public address', () => {
    expect(isPrivateAddress('https://hooks.example.com/mise')).toBe(false);
  });
});
```
