package corpus

import (
	"errors"
	"fmt"
)

var (
	ErrEmbedSpaceMismatch  = errors.New("embed config does not match the locked shared space")
	ErrDuplicateCorpus     = errors.New("corpus ID already registered")
	ErrInvalidMetadataConfig = errors.New("invalid metadata config")
)

func ValidateEmbedSpace(e EmbedConfig) error {
	if e.Model != SharedEmbed.Model {
		return fmt.Errorf("model %q != %q: %w", e.Model, SharedEmbed.Model, ErrEmbedSpaceMismatch)
	}
	if e.Dims != SharedEmbed.Dims {
		return fmt.Errorf("dims %d != %d: %w", e.Dims, SharedEmbed.Dims, ErrEmbedSpaceMismatch)
	}
	if e.TaskType != SharedEmbed.TaskType {
		return fmt.Errorf("task_type %q != %q: %w", e.TaskType, SharedEmbed.TaskType, ErrEmbedSpaceMismatch)
	}
	return nil
}

func Register(d Descriptor) error {
	if _, exists := registry[d.ID]; exists {
		return fmt.Errorf("corpus %s: %w", d.ID, ErrDuplicateCorpus)
	}
	if err := ValidateEmbedSpace(d.Embed); err != nil {
		return err
	}
	registry[d.ID] = d
	return nil
}

func Unregister(id ID) {
	delete(registry, id)
}
```

var ErrGraphRoleViolation = errors.New("graph role does not permit this edge")

func CheckGraphRole(fromCorpusID, toCorpusID string) error {
	fromDesc, ok := corpus.Get(corpus.ID(fromCorpusID))
	if !ok {
		return fmt.Errorf("graph: from_corpus_id %q: %w", fromCorpusID, ErrUnregisteredCorpus)
	}
	if !fromDesc.GraphRole.CanSource {
		return fmt.Errorf("graph: corpus %q cannot source edges: %w", fromCorpusID, ErrGraphRoleViolation)
	}
	toDesc, ok := corpus.Get(corpus.ID(toCorpusID))
	if !ok {
		return fmt.Errorf("graph: to_corpus_id %q: %w", toCorpusID, ErrUnregisteredCorpus)
	}
	if !toDesc.GraphRole.CanTarget {
		return fmt.Errorf("graph: corpus %q cannot be targeted: %w", toCorpusID, ErrGraphRoleViolation)
	}
	return nil
}

package httpapi

import (
	"context"
	"fmt"
	"net/http"

	"github.com/danielgtaylor/huma/v2"

	"danny.vn/mise/pkg/corpus"
)

type CorpusDescriptorWire struct {
	ID             string                `json:"id"`
	Kind           string                `json:"kind"`
	SchemaName     string                `json:"schema_name"`
	CitationScheme string                `json:"citation_scheme"`
	AccessTier     string                `json:"access_tier"`
	Tier           string                `json:"tier,omitempty"`
	Jurisdiction   string                `json:"jurisdiction"`
	EmbedModel     string                `json:"embed_model"`
	EmbedDims      int                   `json:"embed_dims"`
	CanSource      bool                  `json:"can_source"`
	CanTarget      bool                  `json:"can_target"`
}

func descriptorToWire(d corpus.Descriptor) CorpusDescriptorWire {
	return CorpusDescriptorWire{
		ID: string(d.ID), Kind: string(d.Kind),
		SchemaName: d.SchemaName, CitationScheme: d.CitationScheme,
		AccessTier: string(d.AccessTier), Tier: string(d.Tier),
		Jurisdiction: d.Jurisdiction,
		EmbedModel: d.Embed.Model, EmbedDims: d.Embed.Dims,
		CanSource: d.GraphRole.CanSource, CanTarget: d.GraphRole.CanTarget,
	}
}

type RegistryListOutput struct {
	Body struct {
		Items []CorpusDescriptorWire `json:"items"`
	}
}

type RegistryGetInput struct {
	ID string `path:"id"`
}

type RegistryGetOutput struct {
	Body CorpusDescriptorWire
}

func RegisterRegistry(api huma.API) {
	huma.Register(api, huma.Operation{
		OperationID: "list-registry",
		Method:      http.MethodGet,
		Path:        "/registry",
		Summary:     "List all registered corpus descriptors",
		Tags:        []string{"Registry"},
	}, func(_ context.Context, _ *struct{}) (*RegistryListOutput, error) {
		all := corpus.All()
		out := &RegistryListOutput{}
		out.Body.Items = make([]CorpusDescriptorWire, len(all))
		for i, d := range all {
			out.Body.Items[i] = descriptorToWire(d)
		}
		return out, nil
	})

	huma.Register(api, huma.Operation{
		OperationID: "get-registry-corpus",
		Method:      http.MethodGet,
		Path:        "/registry/{id}",
		Summary:     "Get a corpus descriptor by ID",
		Tags:        []string{"Registry"},
		Errors:      []int{http.StatusNotFound},
	}, func(_ context.Context, input *RegistryGetInput) (*RegistryGetOutput, error) {
		d, ok := corpus.Get(corpus.ID(input.ID))
		if !ok {
			return nil, huma.Error404NotFound(fmt.Sprintf("corpus %q not found", input.ID))
		}
		return &RegistryGetOutput{Body: descriptorToWire(d)}, nil
	})
}
```

func (f *fixtureCaptioner) Caption(_ context.Context, image []byte, _ string) (CaptionResult, error) {

	}
	return CaptionResult{
		Text:      string(text),
		Model:     "fixture-vision",
		Prompt:    "Describe this figure.",
		CreatedAt: time.Now().UTC(),
	}, nil

```go
package vertex

import (
	"context"
	"fmt"
	"time"

	genai "cloud.google.com/go/vertexai/genai"
)

type geminiCaptioner struct {
	client *genai.Client
	model  string
}

type geminiCaptionerOpts struct {
	model string
}

func WithCaptionModel(model string) func(*geminiCaptionerOpts) {
	return func(o *geminiCaptionerOpts) { o.model = model }
}

func NewGeminiCaptioner(ctx context.Context, project, region string, opts ...func(*geminiCaptionerOpts)) (Captioner, error) {
	o := &geminiCaptionerOpts{model: "gemini-2.0-flash"}
	for _, fn := range opts {
		fn(o)
	}
	client, err := genai.NewClient(ctx, project, region)
	if err != nil {
		return nil, fmt.Errorf("creating genai client: %w", err)
	}
	return &geminiCaptioner{client: client, model: o.model}, nil
}

func (g *geminiCaptioner) Caption(ctx context.Context, image []byte, contentType string) (CaptionResult, error) {
	model := g.client.GenerativeModel(g.model)
	prompt := "Describe this figure or table from a regulatory compliance document. " +
		"Focus on the structure, data, and regulatory meaning. Be factual and concise."

	resp, err := model.GenerateContent(ctx,
		genai.Blob{MIMEType: contentType, Data: image},
		genai.Text(prompt),
	)
	if err != nil {
		return CaptionResult{}, fmt.Errorf("gemini caption: %w", err)
	}

	var text string
	for _, c := range resp.Candidates {
		if c.Content != nil {
			for _, p := range c.Content.Parts {
				if t, ok := p.(genai.Text); ok {
					text += string(t)
				}
			}
		}
	}
	return CaptionResult{
		Text:      text,
		Model:     g.model,
		Prompt:    prompt,
		CreatedAt: time.Now().UTC(),
	}, nil
}
```

```go
package pipeline

import (
	"context"
	"fmt"

	"github.com/google/uuid"

	"danny.vn/mise/pkg/blob"
	"danny.vn/mise/pkg/rag/embed"
	"danny.vn/mise/pkg/store"
	"danny.vn/mise/pkg/vertex"
)

type CaptionResult struct {
	Section  store.Section
	ImageRef string
}

func CaptionAndEmbed(
	ctx context.Context,
	captioner vertex.Captioner,
	embedder embed.Embedder,
	blobStore blob.Store,
	image []byte,
	contentType string,
	corpusID string,
	docID uuid.UUID,
	figureIndex int,
) (CaptionResult, error) {
	caption, err := captioner.Caption(ctx, image, contentType)
	if err != nil {
		return CaptionResult{}, fmt.Errorf("captioning figure %d: %w", figureIndex, err)
	}

	imageRef := fmt.Sprintf("%s/%s/figure-%d", corpusID, docID, figureIndex)
	if err := blobStore.Put(ctx, imageRef, image); err != nil {
		return CaptionResult{}, fmt.Errorf("storing figure %d image: %w", figureIndex, err)
	}

	vectors, err := embedder.Embed(ctx, []string{caption.Text})
	if err != nil {
		return CaptionResult{}, fmt.Errorf("embedding caption for figure %d: %w", figureIndex, err)
	}

	sec := store.Section{
		ID:          uuid.New(),
		DocumentID:  docID,
		CorpusID:    corpusID,
		HeadingPath: fmt.Sprintf("Figure %d", figureIndex+1),
		Text:        caption.Text,
		ImageRef:    imageRef,
	}
	if len(vectors) > 0 {
		sec.Embedding = vectors[0]
	}

	return CaptionResult{Section: sec, ImageRef: imageRef}, nil
}
```
